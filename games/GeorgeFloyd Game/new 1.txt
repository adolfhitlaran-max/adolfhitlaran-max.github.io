<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Falling Objects Game</title>
    <style>
        body {
            display: flex;
            justify-content: center;
            align-items: center;
            height: 100vh;
            margin: 0;
            background-color: #f0f0f0;
        }
        canvas {
            border: 1px solid #000;
            background-color: #fff;
        }
    </style>
</head>
<body>
    <canvas id="gameCanvas" width="400" height="600"></canvas>

    <script>
        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');

        // Load images
        const playerImg = new Image();
        playerImg.src = 'assets/player.png';

        const goodImgs = [];
        for (let i = 1; i <= 4; i++) {
            const img = new Image();
            img.src = `assets/good${i}.png`;
            goodImgs.push(img);
        }

        const badImgs = [];
        for (let i = 1; i <= 2; i++) {
            const img = new Image();
            img.src = `assets/bad${i}.png`;
            badImgs.push(img);
        }

        // Game variables
        const player = { x: 175, y: 550, width: 50, height: 50, speed: 5 };
        let objects = [];
        let score = 0;
        let hearts = 5;
        let gameRunning = true;
        const keys = {};
        let startTime = Date.now();

        // Key event listeners
        window.addEventListener('keydown', (e) => {
            keys[e.key] = true;
        });

        window.addEventListener('keyup', (e) => {
            keys[e.key] = false;
        });

        // Spawn new object
        function spawnObject() {
            if (!gameRunning) return;
            const isGood = Math.random() < 4/6;
            const imgArray = isGood ? goodImgs : badImgs;
            const img = imgArray[Math.floor(Math.random() * imgArray.length)];
            const obj = {
                x: Math.random() * (canvas.width - 30),
                y: 0,
                width: 30,
                height: 30,
                speed: 2 + Math.random() * 2,
                img: img,
                type: isGood ? 'good' : 'bad'
            };
            objects.push(obj);
        }

        // Game update loop
        function update() {
            if (!gameRunning) return;

            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // Calculate speed factor
            const elapsed = (Date.now() - startTime) / 1000;
            const speedFactor = 1 + elapsed * 0.01; // Increases by 1% per second

            // Move player
            if (keys['ArrowLeft']) player.x -= player.speed;
            if (keys['ArrowRight']) player.x += player.speed;
            if (player.x < 0) player.x = 0;
            if (player.x > canvas.width - player.width) player.x = canvas.width - player.width;

            // Draw player
            ctx.drawImage(playerImg, player.x, player.y, player.width, player.height);

            // Update and draw objects
            for (let i = objects.length - 1; i >= 0; i--) {
                const obj = objects[i];
                obj.y += obj.speed * speedFactor;

                // Draw object
                ctx.drawImage(obj.img, obj.x, obj.y, obj.width, obj.height);

                // Check collision with player
                if (
                    obj.y + obj.height > player.y &&
                    obj.y < player.y + player.height &&
                    obj.x + obj.width > player.x &&
                    obj.x < player.x + player.width
                ) {
                    if (obj.type === 'good') {
                        score += 1;
                    } else {
                        score = Math.max(0, score - 10);
                        hearts--;
                        if (hearts <= 0) {
                            gameRunning = false;
                            alert('Game Over! Final Score: ' + score);
                        }
                    }
                    objects.splice(i, 1);
                } else if (obj.y > canvas.height) {
                    objects.splice(i, 1);
                }
            }

            // Check win condition
            if (score >= 100) {
                gameRunning = false;
                alert('You win the first level! Score: 100');
            }

            // Draw score and hearts
            ctx.font = '20px Arial';
            ctx.fillStyle = '#000';
            ctx.fillText(`Score: ${score}  Hearts: ${hearts}`, 10, 30);

            requestAnimationFrame(update);
        }

        // Start spawning objects every 800ms (adjust for difficulty)
        setInterval(spawnObject, 800);

        // Wait for images to load before starting (simple check for playerImg)
        playerImg.onload = () => {
            update();
        };
    </script>
</body>
</html>
